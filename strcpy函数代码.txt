char *StrCpy(char* src)
{
//1、计算src的长度len
	int i=0,len=0;
	while(src[i]!=0){
	len++;
	i++;
	}
//2、用malloc申请长度为len+1的内存空间
	char* n;
	n=(char* )malloc((len+1)*sizeof(char));
//3、将src字符串逐个字符复制到新申请的内存空间
	for(i=0;i<len;i++){
		n[i]=src[i];
	}
//4、新申请内存空间的最后一位放入'\0'
	n[len]='\0';
//5、返回新申请的内存空间指针
	return n;
}
